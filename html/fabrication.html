Code

Here is the Processing code I used to make these, I made minute changes to the float variables and got a huge variety of results.

import Turtle.*;
import processing.pdf.*;


void setup() {
size (1600, 1600, PDF, "thecutone.pdf");
background(0);
stroke(255);
noLoop();
}


float size = 1.3f;
float thing = 1.03f;
float inc = 1.111;

int count = 0;

void draw(){
thang();
exit();
}

void TurtleCircle(Turtle t, float size, float k) {
for (int i = 0; i < 360; i++) {
//float res = i*1.004;
t.forward(size);
t.right((k*(((int)(i*i))/380)+1));
}
}

void thang() {
background(0);
stroke(255);

Turtle t = new Turtle(this);
t.penUp();
t.goToPoint(300, 320);
t.penDown();


for (int i=0; i<20; i++) {
TurtleCircle(t, size, thing);
t.right(14);
size = size*inc;
}
}
